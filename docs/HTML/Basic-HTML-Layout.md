```html
<html>
  <head>
    <title><!--Appears on the browser window--></title>

    <meta />
    <!--defines metadata about an HTML document. Metadata is data (information)
    about data. -->

    <link rel="icon" href="/favicon.ico" />
    <!--HTML rel attribute is used to specify the relationship between the current
    and the linked document-->
  </head>

  <body>
    <header></header>
    <!-- Header for web page -->

    <main>
      <!-- Main content for web page -->
      <nav>
        <!--Navbar-->
        <a href="#"><!--anchor_tag_1--></a>
      </nav>

      <section>
        <!-- content should be unique to the individual page  -->
        <!-- form -->
        <!-- fieldset  -->
        <!-- legend -->
        <!-- label  -->
        <!-- input  -->
        <!-- button -->
        <!-- etc -->
      </section>

      <div>
        <!-- img -->
        <!-- content -->
        <!-- ul OR ol -->
        <!-- table -->
      </div>
    </main>

    <footer><!-- Footer --></footer>
  </body>
</html>
```
